exports.getUser = function (username) {
    var c = undefined;
    if(!username){username = ""}
    for (var i = 0; i < UserTable.length; i++) {
        if (UserTable[i].username.toLowerCase() == username.toLowerCase()) {
            c = UserTable[i];
            break;
        }
    }
    return c;
};
